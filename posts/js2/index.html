<body>
    <div id="box">
        <p>A</p>
        <p id="para">B</p>
        <p>C</p>
    </div>

    <script>
        var box = document.getElementById('box');
        var para = document.getElementById('para');
        
        console.log(box.childNodes)  // NodeList(7) [text, p, text, p#para, text, p, text]
        console.log(box.children)    // HTMLCollection(3) [p, p#para, p, para: p#para]

        console.log(box.firstChild)         // #text  -> '\n     '
        console.log(box.firstElementChild)  // <p>A</p>
 
        console.log(box.lastChild)          // #text  -> '\n     '
        console.log(box.lastElementChild)   // <p>C</p>

        console.log(para.previousSibling)        // #text  -> '\n     '
        console.log(para.previousElementSibling) // <p>A</p>

        console.log(para.nextSibling)            // #text  -> '\n     '
        console.log(para.nextElementSibling)     // <p>C</p>

        // 一直往上获取，可能会有区别
        console.log(para.parentNode)
        console.log(para.parentElement)
    </script>
</body>

<body>

        <p class="para"> 标志P </p>
    </div>

    <script>
        var box = document.getElementsByClassName('box')[0];
        // 创建节点
        var p1 = document.createElement('p')
        p1.innerText = "insert p"
        var para = document.querySelector('.box .para');  
        // 插入节点      
        box.insertBefore(p1, para)

        var p2 = document.createElement('p')
        p2.innerText = "append p"
        // 追加节点
        box.appendChild(p2);

    </script>

<!-- 鼠标监听 -->
<body>
    <div class="box"></div>
    <script>
        var  box = document.getElementsByClassName('box')[0];
        // 单击
        box.onclick = function() {
            console.log("on click")
        };
        // 双击
        box.ondblclick = function() {
            console.log('double click')
        };
        // 鼠标按下
        box.onmousedown = function() {
            console.log('on mouse down')
        };
        // 鼠标松开
        box.onmouseup = function(){
            console.log('on mouse up')
        }
        // 鼠标移动
        box.onmousemove = function(){
            console.log("on mouse move")
        }
        // 鼠标进入
        box.onmouseenter = function() {
            console.log('on mouse enter')
        }
        // 鼠标离开
        box.onmouseleave = function(){
            console.log('on mouse leave')
        }
    </script>
</body>

<!-- 键盘监听 -->
<body>
    姓名:
    <input type="text" name="username">
    <script>
        var uname = document.getElementsByName("username")[0]
        console.log(uname)

        // 功能键也会响应
        uname.onkeydown = function(){
            console.log('on key down')
        }

        // 按键抬起触发
        uname.onkeyup = function() {
            console.log('on key up')
        }

        // 功能键不会触发
        uname.onkeypress = function() {
            console.log('on key press')
        }
    </script>
</body>

<!-- 表单监听 -->
<body>
    <form id="reg">
    姓名:
    <input type="text" name="username">
    <input type="submit" >
    <input type="reset" >
    </form>
    <script>
        var form = document.getElementById('reg')
        var uname = form.username;
        console.log(uname)

        // 内容改变完了
        uname.onchange = function(){
            console.log('value = ' + uname.value)
        }
        // 内容正在改变
        uname.oninput = function() {
            console.log('value = ' + uname.value)
        }
        // 获取焦点
        uname.onfocus = function() {
            console.log('获取焦点')
        }
        // 失去焦点
        uname.onblur = function() {
            console.log('失去焦点')
        }
        // 提交
        form.onsubmit = function(){
            alert('表单提交')
        }
        // 重置
        form.onreset = function(e){
            alert('表单重置')
        }
    </script>
</body>
