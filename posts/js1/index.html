// ----- string -----
console.log(typeof "hello js");
console.log("hello" + "  world");
var str = "";

// 获取长度
console.log("hello".length);

// 获取索引处字符
console.log("hello".charAt(1));  // e
console.log("hello".charAt(7));  // ""

// substring(a,b) 从a开始到b结束(不包含b)的子串
// b可以省略，则代表截取到结尾
// a可以大于b,会自动调整顺序
// 参数为负数，自动转成0
console.log("hello".substring(0,2));   // he
console.log("hello".substring(3));     // lo
console.log("hello".substring(8,2));   // llo
console.log("hello".substring(-1,3));  // hel
console.log("hello".substring(-1,-4)); // (0,0)""
console.log("hello".substring(-1,4));  // (0,4)hell


// substr(a,b), 从a开始，长度为b
// b可以省略，则截取到结尾
// a可以是负数，表示倒数，或者长度减去a
// b为负数则截取的为空
console.log("hello".substr(1,4));  // ello
console.log("hello".substr(3));    // lo
console.log("hello".substr(-4,3)); // ell

// slice(a,b), 从a开始，b结束，a必须小于b
// a 如果为负数，用长度相减则为起始位置，且满足a < b
console.log("hello".slice(4,2));   // ""
console.log("hello".slice(-1,-3)); // ""
console.log("hello".slice(-3,-1)); //(2,4)ll
console.log("hello".slice(-3,2));  //(2)
console.log("hello".slice(2,3));   // l
console.log("hello".slice(2));     // llo

// 大小写转换
console.log("hello".toUpperCase()); // HELLO
console.log("HELLO".toLowerCase()); // hello

// 获取索引
console.log("HELLO".indexOf("O")); //4

// ----- 转换 -----
// to number
console.log(Number("12"));  // 12
console.log(Number("12hello")); // NaN
console.log(Number(""));    // 0
console.log(Number(true));  // 1
console.log(Number(false)); // 0
console.log(Number(NaN));   // NaN
console.log(Number(undefined)); // NaN
console.log(Number(null));  // 0

console.log(parseInt("10px")); // 10
console.log(parseInt("px10")); // NaN
console.log(parseInt("10.5")); // 10

console.log(parseFloat("10.5px")); // 10.5
console.log(parseFloat("10.99"));  // 10.99

// to string
console.log(String(123));   // "123"
console.log(String(12.3));  // "12.3"
console.log(String(NaN));   // "NaN"
console.log(String(Infinity));  // "Infinity"
console.log(String(undefined)); // "undefined"
console.log(String(null));      // "null"
console.log(String(true));      // "true"
console.log(String(false));     // "false"

console.log(true.toString());   // toString()

// to bool
// 0 & nan = false, other = true
console.log(Boolean(12));  // true
console.log(Boolean(0));   // false
console.log(Boolean(NaN)); // false

console.log(Boolean(""));  // false
console.log(Boolean(undefined)); // false
console.log(Boolean(null));      // false

// 数组
var arr1 = [1,2,3,4,5]
var arr2 = new Array(4)
var arr3 = new Array(1,2,3,4)
console.log(arr1)
console.log(arr2)
console.log(arr3)

// 判断是否是数组
console.log(Array.isArray(arr1))

// 添加元素
arr1.push(1); // 1,2,3,4,5,1
console.log(arr1)

// 删除元素
var item = arr1.pop();  // 1
console.log(arr1);      // 1,2,3,4,5

// 头部插入
arr1.unshift(0,99);  // 0,99,1,2,3,4,5
console.log(arr1);

// 头部删除
var item = arr1.shift(); // 99,1,2,3,4,5
console.log(arr1)

// splice 有返回值，返回值是被删除的项，没被删除则返回[]
// 使用splice替换操作 start, length, items
arr1.splice(0,2,'a','b','c'); //a,b,c,2,3,4,5
console.log(arr1);

// 使用splice插入操作 length = 0
arr1.splice(3,0,1);  // a,b,c,1,2,3,4,5
console.log(arr1);

// 使用splice删除操作，items 不写
arr1.splice(0,3); // 1,2,3,4,5
console.log(arr1);

// slice(a,b), start a , end b
// b可省略，则表示截取到末尾
// a可为负，则表示倒数
var arr11 = arr1.slice(0,3); // 1,2,3
console.log(arr11)
var arr12 = arr1.slice(3); // 4,5
console.log(arr12)
var arr13 = arr1.slice(-2); // 4,5
console.log(arr13)
var arr14 = arr1.slice(-3,-4); //(2,1) 不符合规则了 []
console.log(arr14);

// join
console.log(arr1.join()); //1,2,3,4,5',默认逗号
console.log(arr1.join("")); //'123456'
console.log(arr1.join("-")); //1-2-3-4-5

// split
console.log('12345'.split()); //'12345'
console.log('12345'.split("")) // [1,2,3,4,5]
console.log('1-2-3-4-5'.split("-")); // [1,2,3,4,5]

// concat 
console.log(arr1.concat(arr3)); //1,2,3,4,5,1,2,3,4

// reverse
console.log(arr1.reverse()); // 5,4,3,2,1

// indexof 返回所在位置，没有则-1
// indexof(a,b), a 要找的值，b指定索引开始位置
console.log(arr1.indexOf(5));  // 0
console.log(arr1.indexOf(99)); // -1

// includes 包含某个值
console.log(arr1.includes(5));    // true
console.log(arr1.includes('5'));  // false
console.log(arr1.includes(99));   // false
