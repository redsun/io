<form action="">
    <p>
        用户名:
        <!-- 单行文本域，提交的是value属性值， disabled禁用 -->
        <input type="text" name="username" placeholder="请输入用户名">
    </p>

    <p>
        密码:
        <!-- 单行文本域，type=password 不可见 -->
        <input type="password" placeholder="请输入密码"></p>
    <p>

    <p>
        性别:
        <!-- label将文字和按钮做绑定，点击文字就相当于点击按钮 -->
        <label>
            <!-- 互斥通过name值分为一组 -->
            <input type="radio" name="sex">男
        </label>
        <label>
            <input type="radio" name="sex">女
        </label>
            
    </p>

    <p>
        性别:
        <!-- html4时代，通过for 找input id 对应起来，点击文字相当于点击按钮 -->
        <input type="radio" name="gender" id="boy">
        <label for="boy">男</label>

        <input type="radio" name="gender" id="girl">
        <label for="girl">女</label>
    </p>

    <p>
        想学什么:
        <!-- 通过name 分为一组，提交到服务器是value属性的值, checked默认选中的 -->
        <input type="checkbox" name="learn" value="swift" checked>swift 
        <input type="checkbox" name="learn" value="swiftui" >swiftui
        <input type="checkbox" name="learn" value="combine">combine
    </p>
    
    <p>
        简介:
        <textarea name="" id="" cols="30" rows="10"></textarea>
    </p>

    <!-- HTML5新增 -->
       <p>
        颜色:
        <input type="color">
    </p>

    <p>
        日期:
        <input type="date">
    </p>

    <p>
        时间:
        <input type="time">
    </p>

    <p>
        邮箱:
        <input type="email">
    </p>

    <p>
        文件:
        <input type="file">
    </p>

    <p>
        数字:
        <input type="number" min="1" max="10">
    </p>

    <p>
        进度条:
        <input type="range" min="1" max="10">
    </p>

    <p>
        搜索:
        <input type="search">
    </p>

    <p>
        网址:
        <input type="url">
    </p>

    <p>
        apple:
        <!-- list 和 ID 做绑定，效果就是输入的时候有数据匹配提示 -->
        <input type="text" list="apple-list">
        <datalist id="apple-list">
            <option value="mac pro">
            <option value="mac air">
            <option value="iphone">
            <option value="ipad">
            <option value="ipod">
        </datalist>
    </p>
    <!--  -->
    <select name="region" id="region" multiple>
            <option value="bj">北京</option>
            <option value="tj">天津</option>
            <option value="sh" selected>上海</option>
            <option value="cq">重庆</option>
    </select>

    <p>
        <!-- 按钮，简写为<button> -->
        <input type="button" value="点我一下"> 
        <button>按钮</button>

        <input type="submit"> 
        <!-- 重置按钮 -->
        <input type="reset">

        <input type="image" name="img_button" src="https://imooc.me/assets/img/avatar/avatar.png">

    </p>
