            color: red;
        }
        
        /* 段落5 被选中 */
        .box p:nth-of-type(5) {
            color: yellow;
        }
    </style>
</head>

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>属性选择器</title>
    <style>
        /* alt有值的进行匹配 */
        /* 除了第一个标签其余都被选中 */
        img[alt] {background-color: green;}
        /* alt = 葫芦娃 进行匹配 */
        /* 第二个被选中 */
        img[alt="葫芦娃"] {color: white;}
        /* ^以x开头的进行匹配 */
        /* 2,4,5,6 被选中 */
        img[alt^="葫芦娃"] {font-size: 20px;}
        /* ^ 葫芦娃结尾 */
        /* 2,3，4, 5被选中 */
        img[alt$="葫芦娃"] {font-style: italic;}
        /* ~ 空格隔开 有葫芦娃的 */
        /* 6 被选中 */
        img[alt~="葫芦娃"] {font-family: 'Franklin Gothic Medium', 'Arial Narrow', Arial, sans-serif;}
        /* | 以 北京- 开头的 */
        /* 5 被选中 */
        img[alt|="北京"] {color: blue;}
    </style>
</head>
<body>
    <img src="葫芦娃.jpg">
    <img src="葫芦娃.jpg" alt="葫芦娃">
    <img src="葫芦娃.jpg" alt="葫芦娃北京">
    <img src="葫芦娃.jpg" alt="北京葫芦娃">
    <img src="葫芦娃.jpg" alt="北京-葫芦娃">
    <img src="葫芦娃.jpg" alt="北京 葫芦娃">
</body>
</html>
